	private static String makeParamUrl(Map<String, Object> param) {
		String Str = param.toString();
		Str = Str.replaceAll(",", "&").replaceAll(" ", "").replaceAll("#", " ");
		Str = Str.substring(1);
		Str = Str.substring(0, Str.length() - 1);
		return Str;
	}
